SELECT i.id AS issue_id, il.name AS list_name, i.SUMMARY, i.DESCRIPTION, i.USER_ID_ASSIGNEDTO
FROM ISSUES i
JOIN ISSUELISTS il ON i.ISSUELISTS_ID = il.id;

SELECT i.id AS issue_id, i.SUMMARY, i.DESCRIPTION, u.FIRSTNAME, u.LASTNAME
FROM ISSUES i
JOIN USERS u ON i.USER_ID_ASSIGNEDTO = u.id;

SELECT u.FIRSTNAME, u.LASTNAME, COUNT(i.id) AS task_count
FROM USERS u
LEFT JOIN ISSUES i ON u.id = i.USER_ID_ASSIGNEDTO
GROUP BY u.FIRSTNAME, u.LASTNAME;
